# 1. Dar permiso de ejecución al script
chmod +x ./ejercicio4.ps1

# 2. Iniciar demonio
./ejercicio4.ps1 \
  -directorio "./pruebas/monitoreado" \
  -palabras "password" "account" "unlam" \
  -log "./pruebas/logs/ejercicio4.log"

# 3. Crear archivo con palabra clave
echo "este archivo contiene una password" > ./pruebas/monitoreado/test1.txt

# 4. Modificar archivo agregando otra palabra clave
echo "ahora agrego UNLAM al archivo" >> ./pruebas/monitoreado/test1.txt

# 5. Ver log
cat ./pruebas/logs/ejercicio4.log

# 6. Intentar iniciar otro demonio para el mismo directorio
./ejercicio4.ps1 \
  -directorio "./pruebas/monitoreado" \
  -palabras "password" "account" "unlam" \
  -log "./pruebas/logs/ejercicio4.log"

# 7. Probar monitoreo de subcarpetas
mkdir -p ./pruebas/monitoreado/subcarpeta

echo "archivo dentro de subcarpeta con password" > ./pruebas/monitoreado/subcarpeta/test_sub.txt

# 8. Esperar y ver log

cat ./pruebas/logs/ejercicio4.log

# 9. Detener demonio
./ejercicio4.ps1 -directorio "./pruebas/monitoreado" -kill
